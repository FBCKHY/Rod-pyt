## 🔧 一键清理脚本

### PowerShell完整清理脚本

```powershell
# 项目全面清理脚本
Write-Host "🗑️ 开始全面清理项目..." -ForegroundColor Cyan

$projectDir = "c:\Users\13350\Desktop\oai 08"
Set-Location $projectDir

$deletedCount = 0

# 1. 清理根目录
Write-Host "`n📦 清理根目录..." -ForegroundColor Yellow
$rootFiles = @("cleanup.ps1", "清理建议.md")
foreach ($file in $rootFiles) {
    if (Test-Path $file) {

        Write-Host "  ✅ 已删除: $file" -ForegroundColor Green
        $deletedCount++
    }
}

# 2. 清理报告目录
Write-Host "`n📄 清理报告目录..." -ForegroundColor Yellow
Set-Location "报告"

$reportFiles = @(

    "产品中心开发进度报告.md",
    "产品中心真实完成度报告.md",

    "今日工作总结-2025-11-21.md",
    "今日工作成果展示.md",
    "分类功能修复说明.md",
    "分类功能测试指南.md",
    "分类同步修复说明.md",
    "前端完整现状报告.md",
    "前端开发现状总结.md",
    "后台分类显示问题诊断.md",
    "启动状态-2025-11-21.md",
    "图片404问题修复说明.md",
    "实际完成度核查报告-2025-11-27.md",

    "最终工作总结-完整版.md",
    "最终总结-今日完成情况.md",
    "测试指南.md",
    "测试结果-2025-11-27.md",
    "立即开始行动清单.md",
    "系统启动问题诊断报告.md",
    "系统核实报告-2025-11-27.md",
    "问题解决完成报告.md",
    "项目完成指南-下一步行动.md",
    "项目状态总览-2025-11-21.md",
    "image.png",
    "query"
)

foreach ($file in $reportFiles) {
    if (Test-Path $file) {
        Remove-Item $file -Force
        Write-Host "  ✅ 已删除: $file" -ForegroundColor Green
        $deletedCount++
    }
}

# 3. 删除MD目录
Write-Host "`n📁 清理MD目录..." -ForegroundColor Yellow
if (Test-Path "MD") {
    Remove-Item "MD" -Recurse -Force
    Write-Host "  ✅ 已删除: MD/" -ForegroundColor Green
    $deletedCount++
}

Set-Location $projectDir

# 4. 清空日志文件
Write-Host "`n📝 清空日志文件..." -ForegroundColor Yellow
if (Test-Path "backend\logs\combined.log") {
    "" | Out-File "backend\logs\combined.log"
    Write-Host "  ✅ 已清空: backend\logs\combined.log" -ForegroundColor Green
}
if (Test-Path "backend\logs\error.log") {
    "" | Out-File "backend\logs\error.log"
    Write-Host "  ✅ 已清空: backend\logs\error.log" -ForegroundColor Green
}

# 总结
Write-Host "`n" + "="*50 -ForegroundColor Cyan
Write-Host "✅ 清理完成!" -ForegroundColor Green
Write-Host "📊 删除了 $deletedCount 个文件/目录" -ForegroundColor Cyan
Write-Host "="*50 -ForegroundColor Cyan
```
